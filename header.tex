% Adaptive section TOC: switch to two columns only when subsection count is large.
\newcounter{currsecsubcount}
\newread\fmTocIn
\newcommand{\toccolthreshold}{6}
\newcommand{\countcurrsecsubsections}{%
  \setcounter{currsecsubcount}{0}%
  \IfFileExists{\jobname.toc}{%
    \begingroup
    \edef\fmPatternRaw{subsectionintoc {\number\value{section}}}%
    \edef\fmPattern{\detokenize\expandafter{\fmPatternRaw}}%
    \openin\fmTocIn=\jobname.toc
    \loop\unless\ifeof\fmTocIn
      \read\fmTocIn to \fmLine
      \edef\fmLineSafe{\detokenize\expandafter{\fmLine}}%
      \IfSubStr{\fmLineSafe}{\fmPattern}{\stepcounter{currsecsubcount}}{}
    \repeat
    \closein\fmTocIn
    \endgroup
  }{}
}

\AtBeginSection[]{
  \begin{frame}
    \vfill
    \centering
    \begin{beamercolorbox}[sep=6pt,center,shadow=true,rounded=true]{title}
      \usebeamerfont{title}\LARGE\insertsectionhead\par%
    \end{beamercolorbox}
    \vfill
  \end{frame}
}

\AtBeginSubsection[]{
  \countcurrsecsubsections
  \ifnum\value{currsecsubcount}>\toccolthreshold\relax
    \begin{frame}
      \begingroup
      \usebeamerfont{section in toc}\Large\bfseries\insertsectionhead\par
      \endgroup
      \begin{multicols}{2}
        \tableofcontents[sectionstyle=hide/hide,hideothersubsections, subsectionstyle=show/shaded/hide]
      \end{multicols}
    \end{frame}
  \else
    \begin{frame}
      \begingroup
      \usebeamerfont{section in toc}\large\bfseries\insertsectionhead\par
      \endgroup
      \tableofcontents[sectionstyle=hide/hide,hideothersubsections, subsectionstyle=show/shaded/hide]
    \end{frame}
  \fi
}
